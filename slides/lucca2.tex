\begin{frame}[fragile]

	\begin{columns}
	\column{0.4\textwidth}
		Essa plataforma opera com o seguinte ciclo:
		\begin{enumerate}
			\item Recebimento das percepções;
			\item Raciocínio baseando-se em regras lógicas;
			\item Envio de ordens para os atuadores;
		\end{enumerate}
	\column{0.6\textwidth}
		\begin{center}
		\begin{lstlisting}
		!safety.
		!wandering.

		+!safety
		     : .findall(E, enemy(E), LE)
		     & .length(LE, LEN) & LEN >= 2
		    <- flee; !!safety.
		+!safety <- !!safety.

		+!wandering[source(self)]
		    <- forward; !!wandering.
		-!wandering[source(self)]
		    <- left; !!wandering.

		+enemy(E) <- hit(E).
		\end{lstlisting}
		\end{center}
	\end{columns}

\begin{frame}

	\framesubtitle{Arquitetura Proposta > \emph{Unified} \emph{Modeling} \emph{Language} }
	\begin{itemize}
		\item Relação de Generalização é similar ao \emph{UML}
		\item Relação de Associação direcional representa uma relação entre dois conceitos
		\item Atributos representam relações do conceito para um tipo de dado ou para outros conceitos
		\item Guardas representam condições que precisam ser obedecidas para um elemento pertencer ao conjunto, \\ por exemplo ``int [> 0]''
	\end{itemize}
